            const ruleDiv = document.createElement('div');

        function validateRange(ruleId) {
            const rule = state[ruleId];
            const inputElement = document.getElementById(`value-${ruleId}`);
            const validationElement = document.getElementById(`validation-${ruleId}`);
            const value = parseFloat(inputElement.value);

            if (isNaN(value) || inputElement.value === '') {
                rule.penalty = 0;
                validationElement.textContent = '';
                validationElement.className = 'validation-message';
                updateScores();
                return;
            }

            rule.value = value;
            let violation = 0;
            let message = '';

            if (rule.type === 'range') {
                if (value < rule.min) {
                    // Below minimum
                    violation = Math.ceil(rule.min - value);
                    rule.penalty = violation * rule.basepenalty;
                    message = `Penalty: -${rule.penalty.toFixed(1)} pts`;
                    validationElement.className = 'validation-message fail';
                } else if (value > rule.max) {
                    // Above maximum - round up
                    violation = Math.ceil(value - rule.max);
                    rule.penalty = violation * rule.basepenalty;
                    message = `Penalty: -${rule.penalty.toFixed(1)} pts`;
                    validationElement.className = 'validation-message fail';
                } else {
                    rule.penalty = 0;
                    message = '';
                    validationElement.className = 'validation-message';
                }
            } else if (rule.type === 'minOnly') {
                if (value < rule.min) {
                    violation = Math.ceil(rule.min - value);
                    rule.penalty = violation * rule.basepenalty;
                    message = `Penalty: -${rule.penalty.toFixed(1)} pts`;
                    validationElement.className = 'validation-message fail';
                } else {
                    rule.penalty = 0;
                    message = '';
                    validationElement.className = 'validation-message';
                }
            } else if (rule.type === 'maxOnly') {
                if (value > rule.max) {
                    violation = Math.ceil(value - rule.max);
                    rule.penalty = violation * rule.basepenalty;
                    message = `Penalty: -${rule.penalty.toFixed(1)} pts`;
                    validationElement.className = 'validation-message fail';
                } else {
                    rule.penalty = 0;
                    message = '';
                    validationElement.className = 'validation-message';
                }
            }

            validationElement.textContent = message;
            updateScores();
        }

        function selectAnswer(ruleId, answer) {
            const rule = state[ruleId];
            rule.answer = answer;

            const buttons = document.querySelectorAll(`#${rule.category}-rules .rule-buttons button`);
            buttons.forEach(btn => {
                if (btn.textContent.toLowerCase() === answer && btn.onclick.toString().includes(ruleId)) {
                    btn.classList.add('selected');
                    if (answer === 'no') btn.classList.add('no');
                } else if (btn.onclick.toString().includes(ruleId)) {
                    btn.classList.remove('selected', 'no');
                }
            });

            const inputDiv = document.getElementById(`input-${ruleId}`);
            if (inputDiv && answer === 'no' && (rule.type === 'perWheelInput' || rule.type === 'perWheel')) {
                inputDiv.classList.add('show');
                // Reset penalty until user enters value
                rule.penalty = 0;
            } else if (inputDiv) {
                inputDiv.classList.remove('show');
                if (answer === 'yes') {
                    rule.penalty = 0;
                } else if (answer === 'no') {
                    if (rule.type === 'fixed') {
                        rule.penalty = rule.basepenalty;
                    }
                }
            } else {
                // No input div (fixed rules)
                if (answer === 'yes') {
                    rule.penalty = 0;
                } else if (answer === 'no') {
                    if (rule.type === 'fixed') {
                        rule.penalty = rule.basepenalty;
                    }
                }
            }

            updateScores();
        }

        function calculatePenalty(ruleId) {
            const rule = state[ruleId];
            const inputElement = document.getElementById(`value-${ruleId}`);
            let value = parseFloat(inputElement.value) || 0;
            
            if (rule.type === 'perWheelInput') {
                value = Math.floor(value);
                if (value > 4) value = 4;
                if (value < 0) value = 0;
                inputElement.value = value;
            }
            
            rule.value = value;

            if (rule.answer === 'no' && value > 0) {
                if (rule.type === 'perMMPerWheel') {
                    rule.penalty = value * rule.basepenalty * 4;
                } else if (rule.type === 'perMMPerAxle') {
                    rule.penalty = value * rule.basepenalty;
                } else if (rule.type === 'perWheelInput') {
                    rule.penalty = value * rule.basepenalty;
                } else {
                    rule.penalty = value * rule.basepenalty;
                }
            } else {
                rule.penalty = 0;
            }

            updateScores();
        }

        function updateScores() {
            let generalTotal = 0;
            let safetyTotal = 0;
            let performanceTotal = 0;

            Object.values(state).forEach(rule => {
                if (rule.category === 'general') generalTotal += rule.penalty;
                if (rule.category === 'safety') safetyTotal += rule.penalty;
                if (rule.category === 'performance') performanceTotal += rule.penalty;
            });

            document.getElementById('general-score').textContent = generalTotal.toFixed(1);
            document.getElementById('safety-score').textContent = safetyTotal.toFixed(1);
            document.getElementById('performance-score').textContent = performanceTotal.toFixed(1);

            const totalScore = Math.max(0, 100 - (generalTotal + safetyTotal + performanceTotal));
            document.getElementById('total-score').textContent = totalScore.toFixed(1);
        }

        function exportToCSV() {
            const teamNameInput = document.getElementById('team-name').value;
            
            if (!teamNameInput.trim()) {
                alert('Please enter a team name before submitting!');
                return;
            }

            let generalTotal = 0;
            let safetyTotal = 0;
            let performanceTotal = 0;
            let violations = [];

            Object.values(state).forEach(rule => {
                // Check for violations: either "no" answer with penalty OR numeric value with penalty
                if ((rule.answer === 'no' && rule.penalty > 0) || 
                    (rule.type === 'range' || rule.type === 'minOnly' || rule.type === 'maxOnly') && rule.penalty > 0) {
                    const ruleInfo = regulations.general.concat(regulations.safety, regulations.performance).find(r => {
                        const stateKey = Object.keys(state).find(key => state[key] === rule);
                        return r.id === stateKey;
                    });
                    
                    violations.push({
                        category: rule.category.charAt(0).toUpperCase() + rule.category.slice(1),
                        rule: ruleInfo ? ruleInfo.text : 'Unknown rule',
                        penalty: rule.penalty.toFixed(1)
                    });
                }
                
                if (rule.category === 'general') generalTotal += rule.penalty;
                if (rule.category === 'safety') safetyTotal += rule.penalty;
                if (rule.category === 'performance') performanceTotal += rule.penalty;
            });

            const totalScore = Math.max(0, 100 - (generalTotal + safetyTotal + performanceTotal));
            const currentDate = new Date().toLocaleString();

            // Display violations on page
            displayViolations(teamNameInput, violations, generalTotal, safetyTotal, performanceTotal, totalScore);

            // Create CSV with violations list
            let csvContent = "Team Name,Submission Date,General Penalty,Safety Penalty,Performance Penalty,Total Score\n";
            csvContent += `"${teamNameInput}","${currentDate}",${generalTotal.toFixed(1)},${safetyTotal.toFixed(1)},${performanceTotal.toFixed(1)},${totalScore.toFixed(1)}\n\n`;
            
            if (violations.length > 0) {
                csvContent += "Detected Violations:\n";
                csvContent += "Category,Rule,Penalty\n";
                violations.forEach(v => {
                    csvContent += `"${v.category}","${v.rule}",${v.penalty}\n`;
                });
            } else {
                csvContent += "\nNo violations detected - Perfect compliance!\n";
            }

            const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
            const link = document.createElement("a");
            const url = URL.createObjectURL(blob);
            
            link.setAttribute("href", url);
            link.setAttribute("download", `STEM_Racing_Score_${teamNameInput.replace(/\s+/g, '_')}_${Date.now()}.csv`);
            link.style.visibility = 'hidden';
            
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);

            // Scroll to violations section
            document.getElementById('violations-section').scrollIntoView({ behavior: 'smooth', block: 'start' });

            alert(`Score submitted successfully!\nCSV file has been downloaded.\n\nScroll down to see the detailed inspection results.`);
        }

        function displayViolations(teamName, violations, generalTotal, safetyTotal, performanceTotal, totalScore) {

            const violationsList = document.getElementById('violations-list');

            document.getElementById('result-team-name').textContent = teamName;

            document.getElementById('result-safety').textContent = safetyTotal.toFixed(1);
            document.getElementById('result-performance').textContent = performanceTotal.toFixed(1);
            document.getElementById('result-total').textContent = totalScore.toFixed(1);
            
            // Clear previous violations
            violationsList.innerHTML = '';
            
            if (violations.length > 0) {
                violations.forEach(v => {
                    const violationDiv = document.createElement('div');
                    violationDiv.className = 'violation-item';
                    violationDiv.innerHTML = `
                        <div class="violation-category ${v.category.toLowerCase()}">${v.category}</div>

                        <div class="violation-penalty">Penalty: -${v.penalty} pts</div>

                    violationsList.appendChild(violationDiv);
                });
            } else {
                const noViolationsDiv = document.createElement('div');
                noViolationsDiv.className = 'no-violations';
                noViolationsDiv.textContent = 'No violations detected! Perfect compliance!';
                violationsList.appendChild(noViolationsDiv);
            }
            
            // Show the violations section
            violationsSection.classList.add('show');
        }
